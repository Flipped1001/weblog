```sql
-- LeetCode 
-- 1527.患有某种疾病的患者

SELECT 
    patient_id,
    patient_name,
    conditions
FROM 
    Patients
WHERE 
    conditions LIKE 'DIAB1%' OR conditions LIKE '% DIAB1%';
    
-- 用LIKE查找含有DIAB1为前缀的conditions
```
