# 函数

str1="hsdhaui"
str2="nmksadsi"
str3="dnsajknd"

def my_len (data):
    # 可以给函数加一个说明文档，方便后续应用
       """
    函数说明
    :param data:形参
    :return:data的长度
    """
    
    count = 0
    for i in data:
        count += 1
    return count

num1=my_len(str1)
num2=my_len(str2)
num3=my_len(str3)

print(num1,num2,num3)
